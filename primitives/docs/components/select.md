 export default () =&gt; (   &lt;Select.Root&gt;     &lt;Select.Trigger&gt;       &lt;Select.Value /&gt;        &lt;Select.Icon /&gt;     &lt;/Select.Trigger&gt;      &lt;Select.Portal&gt;       &lt;Select.Content&gt;         &lt;Select.ScrollUpButton /&gt;          &lt;Select.Viewport&gt;           &lt;Select.Item&gt;             &lt;Select.ItemText /&gt;              &lt;Select.ItemIndicator /&gt;           &lt;/Select.Item&gt;            &lt;Select.Group&gt;             &lt;Select.Label /&gt;              &lt;Select.Item&gt;               &lt;Select.ItemText /&gt;                &lt;Select.ItemIndicator /&gt;             &lt;/Select.Item&gt;     &lt;/Select.Portal&gt;   &lt;/Select.Root&gt; ); ```  ## [API Reference](#api-reference)  ### [Root](#root)  Contains all the parts of a select.  | Prop            | Type       | Default          | | --------------- | ---------- | ---------------- | | `defaultValue`  | `string`   | No default value | | `value`         | `string`   | No default value | | `onValueChange` | `function` | No default value | | `defaultOpen`   | `boolean`  | No default value | | `open`          | `boolean`  | No default value | | `onOpenChange`  | `function` | No default value | | `dir`           | `enum`     | No default value | | `name`          | `string`   | No default value | | `disabled`      | `boolean`  | No default value | | `required`      | `boolean`  | No default value |  ### [Trigger](#trigger)  The button that toggles the select. If you would prefer an alternative positioning approach similar to `Popover` or `DropdownMenu` then you can set `position` to `popper` and make use of additional alignment options such as `side`, `sideOffset` and more.  ```jsx // index.jsx  import { Select } from "radix-ui";  export default () =&gt; (   &lt;Select.Root&gt;     &lt;Select.Trigger&gt;…&lt;/Select.Trigger&gt;      &lt;Select.Portal&gt;       &lt;Select.Content position="popper" sideOffset={5}&gt;         …       &lt;/Select.Content&gt;     &lt;/Select.Portal&gt;   &lt;/Select.Root&gt; ); ```  ### [Constrain the content size](#constrain-the-content-size)  When using `position="popper"` on `Select.Content`, you may want to constrain the width of the content so that it matches the trigger width. You may also want to constrain its height to not exceed the viewport.  We expose several CSS custom properties such as `--radix-select-trigger-width` and `--radix-select-content-available-height` to support this. Use them to constrain the content dimensions.  ```jsx // index.jsx  import { Select } from "radix-ui";  import "./styles.css";  export default () =&gt; (   &lt;Select.Root&gt;     &lt;Select.Trigger&gt;…&lt;/Select.Trigger&gt;      &lt;Select.Portal&gt;       &lt;Select.Content className="SelectContent" position="popper" sideOffset={5}&gt;         …       &lt;/Select.Content&gt;     &lt;/Select.Portal&gt;   &lt;/Select.Root&gt; ); ```  ```css /* styles.css */  .SelectContent {   width: var(--radix-select-trigger-width);    max-height: var(--radix-select-content-available-height); } ```  ### [With disabled items](#with-disabled-items)  You can add special styles to disabled items via the `data-disabled` attribute.  ```jsx // index.jsx  import { Select } from "radix-ui";  import "./styles.css";  export default () =&gt; (   &lt;Select.Root&gt;     &lt;Select.Trigger&gt;…&lt;/Select.Trigger&gt;      &lt;Select.Portal&gt;       &lt;Select.Content&gt;         &lt;Select.Viewport&gt;           &lt;Select.Item className="SelectItem" disabled&gt;             …           &lt;/Select.Item&gt;            &lt;Select.Item&gt;…&lt;/Select.Item&gt;            &lt;Select.Item&gt;…&lt;/Select.Item&gt;         &lt;/Select.Viewport&gt;       &lt;/Select.Content&gt;     &lt;/Select.Portal&gt;   &lt;/Select.Root&gt; ); ```  ```css /* styles.css */  .SelectItem[data-disabled] {   color: "gainsboro"; } ```  ### [With a placeholder](#with-a-placeholder)  You can use the `placeholder` prop on `Value` for when the select has no value. There's also a `data-placeholder` attribute on `Trigger` to help with styling.  ```jsx // index.jsx  import { Select } from "radix-ui";  import "./styles.css";  export default () =&gt; (   &lt;Select.Root&gt;     &lt;Select.Trigger className="SelectTrigger"&gt;       &lt;Select.Value placeholder="Pick an option" /&gt;        &lt;Select.Icon /&gt;     &lt;/Select.Trigger&gt;      &lt;Select.Portal&gt;       &lt;Select.Content&gt;…&lt;/Select.Content&gt;     &lt;/Select.Portal&gt;   &lt;/Select.Root&gt; ); ```  ```css /* styles.css */  .SelectTrigger[data-placeholder] {   color: "gainsboro"; } ```  ### [With separators](#with-separators)  Use the `Separator` part to add a separator between items.  ```jsx &lt;Select.Root&gt;   &lt;Select.Trigger&gt;…&lt;/Select.Trigger&gt;    &lt;Select.Portal&gt;     &lt;Select.Content&gt;       &lt;Select.Viewport&gt;         &lt;Select.Item&gt;…&lt;/Select.Item&gt;          &lt;Select.Item&gt;…&lt;/Select.Item&gt;          &lt;Select.Item&gt;…&lt;/Select.Item&gt;          &lt;Select.Separator /&gt;          &lt;Select.Item&gt;…&lt;/Select.Item&gt;          &lt;Select.Item&gt;…&lt;/Select.Item&gt;       &lt;/Select.Viewport&gt;     &lt;/Select.Content&gt;   &lt;/Select.Portal&gt; &lt;/Select.Root&gt; ```  ### [With grouped items](#with-grouped-items)  Use the `Group` and `Label` parts to group items in a section.  ```jsx &lt;Select.Root&gt;   &lt;Select.Trigger&gt;…&lt;/Select.Trigger&gt;    &lt;Select.Portal&gt;     &lt;Select.Content&gt;       &lt;Select.Viewport&gt;         &lt;Select.Group&gt;           &lt;Select.Label&gt;Label&lt;/Select.Label&gt;            &lt;Select.Item&gt;…&lt;/Select.Item&gt;            &lt;Select.Item&gt;…&lt;/Select.Item&gt;            &lt;Select.Item&gt;…&lt;/Select.Item&gt;         &lt;/Select.Group&gt;       &lt;/Select.Viewport&gt;     &lt;/Select.Content&gt;   &lt;/Select.Portal&gt; &lt;/Select.Root&gt; ```  ### [With complex items](#with-complex-items)  You can use custom content in your items.  ```jsx import { Select } from "radix-ui";  export default () =&gt; (   &lt;Select.Root&gt;     &lt;Select.Trigger&gt;…&lt;/Select.Trigger&gt;      &lt;Select.Portal&gt;       &lt;Select.Content&gt;         &lt;Select.Viewport&gt;           &lt;Select.Item&gt;             &lt;Select.ItemText&gt;               &lt;img src="…" /&gt;               Adolfo Hess             &lt;/Select.ItemText&gt;              &lt;Select.ItemIndicator&gt;…&lt;/Select.ItemIndicator&gt;           &lt;/Select.Item&gt;            &lt;Select.Item&gt;…&lt;/Select.Item&gt;            &lt;Select.Item&gt;…&lt;/Select.Item&gt;         &lt;/Select.Viewport&gt;       &lt;/Select.Content&gt;     &lt;/Select.Portal&gt;   &lt;/Select.Root&gt; ); ```  ### [Controlling the value displayed in the trigger](#controlling-the-value-displayed-in-the-trigger)  By default the trigger will automatically display the selected item `ItemText`'s content. You can control what appears by chosing to put things inside/outside the `ItemText` part.  If you need more flexibility, you can control the component using `value`/`onValueChange` props and passing `children` to `SelectValue`. Remember to make sure what you put in there is accessible.  ```jsx const countries = { france: "🇫🇷", "united-kingdom": "🇬🇧", spain: "🇪🇸" };  export default () =&gt; {   const [value, setValue] = React.useState("france");    return (     &lt;Select.Root value={value} onValueChange={setValue}&gt;       &lt;Select.Trigger&gt;         &lt;Select.Value aria-label={value}&gt;{countries[value]}&lt;/Select.Value&gt;          &lt;Select.Icon /&gt;       &lt;/Select.Trigger&gt;        &lt;Select.Portal&gt;         &lt;Select.Content&gt;           &lt;Select.Viewport&gt;             &lt;Select.Item value="france"&gt;               &lt;Select.ItemText&gt;France&lt;/Select.ItemText&gt;                &lt;Select.ItemIndicator&gt;…&lt;/Select.ItemIndicator&gt;             &lt;/Select.Item&gt;              &lt;Select.Item value="united-kingdom"&gt;               &lt;Select.ItemText&gt;United Kingdom&lt;/Select.ItemText&gt;                &lt;Select.ItemIndicator&gt;…&lt;/Select.ItemIndicator&gt;             &lt;/Select.Item&gt;              &lt;Select.Item value="spain"&gt;               &lt;Select.ItemText&gt;Spain&lt;/Select.ItemText&gt;                &lt;Select.ItemIndicator&gt;…&lt;/Select.ItemIndicator&gt;             &lt;/Select.Item&gt;           &lt;/Select.Viewport&gt;         &lt;/Select.Content&gt;       &lt;/Select.Portal&gt;     &lt;/Select.Root&gt;   ); }; ```  ### [With custom scrollbar](#with-custom-scrollbar)  The native scrollbar is hidden by default as we recommend using the `ScrollUpButton` and `ScrollDownButton` parts for the best UX. If you do not want to use these parts, compose your select with our [Scroll Area](/primitives/docs/components/scroll-area) primitive.  ```jsx // index.jsx  import { Select, ScrollArea } from "radix-ui";  import "./styles.css";  export default () =&gt; (   &lt;Select.Root&gt;     &lt;Select.Trigger&gt;…&lt;/Select.Trigger&gt;      &lt;Select.Portal&gt;       &lt;Select.Content&gt;         &lt;ScrollArea.Root className="ScrollAreaRoot" type="auto"&gt;           &lt;Select.Viewport asChild&gt;             &lt;ScrollArea.Viewport className="ScrollAreaViewport"&gt;               &lt;StyledItem&gt;…&lt;/StyledItem&gt;                &lt;StyledItem&gt;…&lt;/StyledItem&gt;                &lt;StyledItem&gt;…&lt;/StyledItem&gt;             &lt;/ScrollArea.Viewport&gt;           &lt;/Select.Viewport&gt;            &lt;ScrollArea.Scrollbar className="ScrollAreaScrollbar" orientation="vertical"&gt;             &lt;ScrollArea.Thumb className="ScrollAreaThumb" /&gt;           &lt;/ScrollArea.Scrollbar&gt;         &lt;/ScrollArea.Root&gt;       &lt;/Select.Content&gt;     &lt;/Select.Portal&gt;   &lt;/Select.Root&gt; ); ```  ```css /* styles.css */  .ScrollAreaRoot {   width: 100%;    height: 100%; }  .ScrollAreaViewport {   width: 100%;    height: 100%; }  .ScrollAreaScrollbar {   width: 4px;    padding: 5px 2px; }  .ScrollAreaThumb {   background: rgba(0, 0, 0, 0.3);    border-radius: 3px; } ```  ## [Accessibility](#accessibility)  Adheres to the [ListBox WAI-ARIA design pattern](https://www.w3.org/WAI/ARIA/apg/patterns/listbox).  See the W3C [Select-Only Combobox](https://www.w3.org/TR/wai-aria-practices/examples/combobox/combobox-select-only.html) example for more information.  ### [Keyboard Interactions](#keyboard-interactions)  | Key         | Description                                                                                                                            | | ----------- | -------------------------------------------------------------------------------------------------------------------------------------- | | `Space`     | When focus is on `Select.Trigger`, opens the select and focuses the selected item.  export default () =&gt; (   &lt;&gt;     &lt;Label&gt;       Country       &lt;Select.Root&gt;…&lt;/Select.Root&gt;     &lt;/Label&gt;      {/* or */}      &lt;Label htmlFor="country"&gt;Country&lt;/Label&gt;      &lt;Select.Root&gt;       &lt;Select.Trigger id="country"&gt;…&lt;/Select.Trigger&gt;        &lt;Select.Portal&gt;         &lt;Select.Content&gt;…&lt;/Select.Content&gt;       &lt;/Select.Portal&gt;     &lt;/Select.Root&gt;   &lt;/&gt; ); ```  ## [Custom APIs](#custom-apis)  Create your own API by abstracting the primitive parts into your own component.  ### [Abstract down to `Select` and `SelectItem`](#abstract-down-to-select-and-selectitem)  This example abstracts most of the parts.  #### Usage  ```jsx import { Select, SelectItem } from "./your-select";  export default () =&gt; (   &lt;Select defaultValue="2"&gt;     &lt;SelectItem value="1"&gt;Item 1&lt;/SelectItem&gt;      &lt;SelectItem value="2"&gt;Item 2&lt;/SelectItem&gt;      &lt;SelectItem value="3"&gt;Item 3&lt;/SelectItem&gt;  export const Select = React.forwardRef(({ children, ...props }, forwardedRef) =&gt; {   return (     &lt;SelectPrimitive.Root {...props}&gt;       &lt;SelectPrimitive.Trigger ref={forwardedRef}&gt;         &lt;SelectPrimitive.Value /&gt;          &lt;SelectPrimitive.Icon&gt;           &lt;ChevronDownIcon /&gt;         &lt;/SelectPrimitive.Icon&gt;       &lt;/SelectPrimitive.Trigger&gt;        &lt;SelectPrimitive.Portal&gt;         &lt;SelectPrimitive.Content&gt;           &lt;SelectPrimitive.ScrollUpButton&gt;             &lt;ChevronUpIcon /&gt;           &lt;/SelectPrimitive.ScrollUpButton&gt;            &lt;SelectPrimitive.Viewport&gt;{children}&lt;/SelectPrimitive.Viewport&gt;     &lt;/SelectPrimitive.Root&gt;   ); });  export const SelectItem = React.forwardRef(({ children, ...props }, forwardedRef) =&gt; {   return (     &lt;SelectPrimitive.Item {...props} ref={forwardedRef}&gt;       &lt;SelectPrimitive.ItemText&gt;{children}&lt;/SelectPrimitive.ItemText&gt;        &lt;SelectPrimitive.ItemIndicator&gt;         &lt;CheckIcon /&gt;       &lt;/SelectPrimitive.ItemIndicator&gt;     &lt;/SelectPrimitive.Item&gt;   ); }); ```
